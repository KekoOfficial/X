async function saveNexus() { const btn = document.querySelector('.save-btn'); const data = { bot_token: document.getElementById('bot-token').value, chat_id: document.getElementById('chat-id').value }; // Feedback visual inmediato btn.innerText = "GUARDANDO..."; btn.style.opacity = "0.7"; try { const res = await fetch("/api/save_settings", { method: "POST", headers: {"Content-Type": "application/json"}, body: JSON.stringify(data) }); if(res.ok) { btn.innerText = "¡GUARDADO!"; btn.style.background = "#00ff88"; // Verde éxito // Esperar 1 segundo para que veas el éxito y luego cerrar setTimeout(() => { // Intentar volver a la consola principal window.location.href = "/"; }, 1000); } else { alert("❌ Error al guardar en la base de datos."); btn.innerText = "ACTUALIZAR BASE DE DATOS"; btn.style.opacity = "1"; } } catch (err) { alert("❌ Error de conexión con la Consola."); btn.innerText = "ACTUALIZAR BASE DE DATOS"; } }
